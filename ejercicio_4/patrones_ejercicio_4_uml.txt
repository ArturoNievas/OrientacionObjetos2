@startuml

abstract Empleado{
	+sueldo(): Real
	+descuento(): Real
 	+{abstract} basico(): Real
	+{abstract} adicional(): Real
}

class Temporario{
	-casado: Boolean
 	-hijos: Integer
	+basico(): Real
	+adicional(): Real
}

class Pasante{
	-examenes: Integer
	+basico(): Real
	+adicional(): Real
}

class Planta{
	-casado: Boolean
 	-hijos: Integer
 	-antiguedad: Integer
	+basico(): Real
	+adicional(): Real
}

Empleado <|-- Temporario
Empleado <|-- Pasante
Empleado <|-- Planta


@enduml
